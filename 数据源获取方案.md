### 4.2 数据源管理器实现

```python
# trendradar/data/source_manager.py
"""
数据源管理器 - 配置驱动的数据获取系统
"""

import yaml
from typing import Dict, List, Optional, Any
from dataclasses import dataclass
from datetime import datetime
from abc import ABC, abstractmethod
import pandas as pd

# 动态导入AKShare（如果未安装则跳过）
try:
    import akshare as ak
    AKSHARE_AVAILABLE = True
except ImportError:
    AKSHARE_AVAILABLE = False
    print("警告: AKShare未安装，请运行: pip install akshare")

import requests
from bs4 import BeautifulSoup


@dataclass
class DataPoint:
    """数据点"""
    source_id: str          # 数据源ID
    date: datetime          # 数据日期
    value: float            # 数值
    category: str           # 数据类别（price/production/inventory...）
    type: str               # 数据类型（spot/future/exchange...）
    product: str            # 产品（steel/copper/aluminum...）
    metadata: Dict = None   # 额外信息


class DataSourceBase(ABC):
    """数据源基类"""

    def __init__(self, config: Dict):
        self.config = config
        self.source_id = ""
        self.name = config.get("name", "")
        self.enabled = config.get("enabled", True)
        self.data_category = config.get("data_category", "")
        self.data_type = config.get("data_type", "")
        self.product = config.get("product", "")

    @abstractmethod
    def fetch(self) -> List[DataPoint]:
        """获取数据"""
        pass


class AKShareDataSource(DataSourceBase):
    """AKShare数据源"""

    def __init__(self, source_id: str, config: Dict):
        super().__init__(config)
        self.source_id = source_id
        self.function = config.get("function", "")
        self.params = config.get("params", {})
        self.field_mapping = config.get("field_mapping", {})

    def fetch(self) -> List[DataPoint]:
        """获取AKShare数据"""
        if not AKSHARE_AVAILABLE:
            raise ImportError("AKShare未安装")

        # 动态调用AKShare函数
        func = getattr(ak, self.function)
        df = func(**self.params)

        # 转换为DataPoint列表
        data_points = []

        # 识别日期列
        date_field = self.field_mapping.get("date", "日期")
        value_field = self.field_mapping.get("close", "收盘价")

        for _, row in df.iterrows():
            try:
                # 解析日期
                date_str = str(row.get(date_field, ""))
                date = pd.to_datetime(date_str)

                # 解析数值
                value = float(row.get(value_field, 0))

                data_point = DataPoint(
                    source_id=self.source_id,
                    date=date,
                    value=value,
                    category=self.data_category,
                    type=self.data_type,
                    product=self.product,
                    metadata={
                        "raw_data": row.to_dict()
                    }
                )
                data_points.append(data_point)

            except Exception as e:

class WebScrapeDataSource(DataSourceBase):
    """网页爬取数据源"""

    def __init__(self, source_id: str, config: Dict):
        super().__init__(config)
        self.source_id = source_id
        self.url = config.get("url", "")
        self.method = config.get("method", "GET")
        self.encoding = config.get("encoding", "utf-8")
        self.selectors = config.get("selectors", {})
        self.cleaning = config.get("cleaning", {})

    def fetch(self) -> List[DataPoint]:
        """爬取网页数据"""
        # 发起请求
        response = requests.get(self.url, timeout=30)
        response.encoding = self.encoding

        # 解析HTML
        soup = BeautifulSoup(response.text, 'html.parser')

        # 查找表格
        table = soup.select_one(self.selectors.get("table", "table"))
        if not table:
            print(f"[{self.name}] 未找到数据表格")
            return []

        # 解析行
        rows = table.select(self.selectors.get("rows", "tr"))

        data_points = []
        for row in rows[1:]:  # 跳过表头
            cells = row.find_all("td")
            if len(cells) < 2:
                continue

            try:
                # 提取数据（根据columns配置）
                columns = self.selectors.get("columns", [])

                date_text = cells[0].get_text(strip=True)
                value_text = cells[1].get_text(strip=True)

                # 数据清洗
                remove_chars = self.cleaning.get("remove_chars", [])
                for char in remove_chars:
                    value_text = value_text.replace(char, "")

                # 转换类型
                date = pd.to_datetime(date_text)
                value = float(value_text)

                data_point = DataPoint(
                    source_id=self.source_id,
                    date=date,
                    value=value,
                    category=self.data_category,
                    type=self.data_type,
                    product=self.product
                )
                data_points.append(data_point)

            except Exception as e:
                print(f"[{self.name}] 行解析失败: {e}")
                continue

        print(f"[{self.name}] 成功爬取 {len(data_points)} 条数据")
        return data_points

    def _replace_env_vars(self, obj: Any) -> Any:
        """替换环境变量"""
        import os
        import re

        if isinstance(obj, dict):
            return {k: self._replace_env_vars(v) for k, v in obj.items()}
        elif isinstance(obj, str):
            # 替换 ${VAR_NAME} 格式
            pattern = r'\$\{(\w+)\}'

                lambda m: os.environ.get(m.group(1), m.group(0)),
                obj
            )
        else:
            return obj


class DataSourceManager:
    """数据源管理器"""

    def __init__(self, config_path: str):
        self.config_path = config_path
        self.sources: Dict[str, DataSourceBase] = {}
        self._load_config()

    def _load_config(self):
        """加载配置文件"""
        with open(self.config_path, 'r', encoding='utf-8') as f:
            config = yaml.safe_load(f)

        data_sources = config.get("data_sources", {})

        for source_id, source_config in data_sources.items():
            if not source_config.get("enabled", True):
                continue

            source_type = source_config.get("type", "")

            # 根据类型创建数据源实例
            if source_type == "akshare":
                source = AKShareDataSource(source_id, source_config)
            elif source_type == "web_scrape":
                source = WebScrapeDataSource(source_id, source_config)
            elif source_type == "api":
                source = APIDataSource(source_id, source_config)
            else:
                print(f"未知数据源类型: {source_type}")
                continue

            self.sources[source_id] = source

        print(f"✓ 加载了 {len(self.sources)} 个数据源")

    def fetch_all(self) -> Dict[str, List[DataPoint]]:
        """获取所有数据源的数据"""
        results = {}

        for source_id, source in self.sources.items():
            try:
                print(f"\n[{source.name}] 开始获取数据...")
                data_points = source.fetch()
                results[source_id] = data_points
            except Exception as e:
                print(f"[{source.name}] 获取失败: {e}")
                results[source_id] = []

        return results

    def fetch_one(self, source_id: str) -> List[DataPoint]:
        """获取单个数据源的数据"""
        source = self.sources.get(source_id)
        if not source:
            raise ValueError(f"数据源不存在: {source_id}")

        return source.fetch()

    def list_sources(self) -> List[Dict]:
        """列出所有数据源"""
        return [
            {
                "id": source_id,
                "name": source.name,
                "type": source.config.get("type"),
                "category": source.data_category,
                "product": source.product,
                "enabled": source.enabled
            }
            for source_id, source in self.sources.items()
        ]
```
