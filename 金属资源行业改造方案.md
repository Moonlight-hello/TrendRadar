## 改造目标

### 核心需求
针对**金属资源行业**（铜、铝、锌、镍等），实现：

1. **多维度数据采集**
   - 新闻热点：原有功能保持
   - 进出口数据：海关总署、各国海关数据
   - 供需数据：产量、消费量、库存
   - 价格数据：期货价格、现货价格、升贴水
   - 宏观数据：GDP、PMI、固定资产投资

2. **智能关联分析**
   - 当出现"铜价上涨"新闻时，自动关联：
     - ✅ 需求端：下游消费（电力、家电、汽车等）
     - ✅ 供给端：矿山产量、冶炼厂产能、进口量
     - ✅ 库存变化：交易所库存、保税区库存、社会库存
     - ✅ 异常检测：识别囤货、逼仓等异常行为

3. **可视化报告**
   - 新闻+数据综合分析
   - 供需平衡表
   - 价格驱动因素拆解

## 爬虫模块设计

### 1. 数据源基类：`trendradar/crawler/data_sources/base.py`

```python
"""
数据源爬虫基类
"""

from abc import ABC, abstractmethod
from typing import List, Dict, Optional, Any
from dataclasses import dataclass
from datetime import datetime
from enum import Enum

import requests
from tenacity import retry, stop_after_attempt, wait_exponential


class DataSourceType(Enum):
    """数据源类型"""
    HTML_SCRAPE = "html_scrape"  # HTML网页抓取
    API = "api"                   # API接口
    PDF_REPORT = "pdf_report"     # PDF报告
    EXCEL = "excel"               # Excel文件
    CSV = "csv"                   # CSV文件


@dataclass
class MetalDataPoint:
    """金属数据点"""
    metal: str              # 金属名称（copper, aluminum等）
    data_type: str          # 数据类型（import, export, production, consumption等）
    value: float            # 数值
    unit: str               # 单位（ton, kt等）
    date: datetime          # 数据日期
    source: str             # 数据源
    region: str = "China"   # 地区
    metadata: Dict = None   # 额外元数据


class DataSourceBase(ABC):
    """数据源爬虫基类"""

    def __init__(
        self,
        name: str,
        url: str,
        source_type: DataSourceType,
        timeout: int = 30,
        retry_times: int = 3,
        use_proxy: bool = False,
        proxy_url: str = "",
    ):
        self.name = name
        self.url = url
        self.source_type = source_type
        self.timeout = timeout
        self.retry_times = retry_times

        # 创建session
        self.session = requests.Session()
        self.session.headers.update({
            "User-Agent": "TrendRadar/2.0 Data Collector",
            "Accept": "*/*",
        })

        if use_proxy and proxy_url:
            self.session.proxies = {
                "http": proxy_url,
                "https": proxy_url,
            }

    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=4, max=10)
    )
    def fetch_raw_data(self, url: str) -> str:
        """
        获取原始数据

        Args:
            url: 目标URL

        Returns:
            原始数据（HTML/JSON等）
        """
        response = self.session.get(url, timeout=self.timeout)
        response.raise_for_status()
        return response.text

    @abstractmethod
    def parse(self, raw_data: str) -> List[MetalDataPoint]:
        """
        解析原始数据为标准格式

        Args:
            raw_data: 原始数据

        Returns:
            数据点列表
        """
        pass

    def fetch_and_parse(self) -> List[MetalDataPoint]:
        """
        抓取并解析数据

        Returns:
            数据点列表
        """
        print(f"[{self.name}] 开始抓取数据...")
        raw_data = self.fetch_raw_data(self.url)

        print(f"[{self.name}] 开始解析数据...")
        data_points = self.parse(raw_data)

        print(f"[{self.name}] 完成，共获取 {len(data_points)} 条数据")
        return data_points

    def standardize_unit(self, value: float, from_unit: str, to_unit: str = "ton") -> float:
        """
        单位标准化

        Args:
            value: 原始数值
            from_unit: 原始单位
            to_unit: 目标单位

        Returns:
            转换后的数值
        """
        # 单位转换表（全部转为吨）
        unit_factors = {
            "ton": 1,
            "t": 1,
            "mt": 1,
            "kg": 0.001,
            "kt": 1000,
            "万吨": 10000,
        }

        from_factor = unit_factors.get(from_unit.lower(), 1)
        to_factor = unit_factors.get(to_unit.lower(), 1)

        return value * from_factor / to_factor
```

### 2. 海关数据爬虫示例：`trendradar/crawler/data_sources/customs.py`

```python
"""
海关进出口数据爬虫
"""

from typing import List
from datetime import datetime
from bs4 import BeautifulSoup
import re

from .base import DataSourceBase, DataSourceType, MetalDataPoint


class ChinaCustomsCrawler(DataSourceBase):
    """中国海关总署数据爬虫"""

    def __init__(self, config: dict):
        super().__init__(
            name=config.get("name", "中国海关总署"),
            url=config.get("url"),
            source_type=DataSourceType.HTML_SCRAPE,
            timeout=config.get("timeout", 30),
        )
        self.selectors = config.get("selectors", {})
        self.encoding = config.get("encoding", "utf-8")

    def parse(self, raw_data: str) -> List[MetalDataPoint]:
        """
        解析海关网页数据

        HTML结构示例：
        <table class="data-table">
            <tr>
                <td>铜及铜制品</td>
                <td>450000</td>  <!-- 进口量，吨 -->
                <td>50000</td>   <!-- 出口量，吨 -->
                <td>2025-01</td>
            </tr>
        </table>
        """
        soup = BeautifulSoup(raw_data, 'html.parser')
        data_points = []

        # 查找数据表
        table = soup.select_one(self.selectors.get("table", "table"))
        if not table:
            print(f"[{self.name}] 未找到数据表")
            return data_points

        # 遍历数据行
        rows = table.select(self.selectors.get("rows", "tr"))

        for row in rows[1:]:  # 跳过表头
            cells = row.find_all("td")
            if len(cells) < 4:
                continue

            try:
                # 提取数据
                product_name = cells[0].get_text(strip=True)
                import_value = self._parse_number(cells[1].get_text(strip=True))
                export_value = self._parse_number(cells[2].get_text(strip=True))
                date_str = cells[3].get_text(strip=True)

                # 解析日期
                date = self._parse_date(date_str)

                # 判断金属类型
                metal = self._identify_metal(product_name)
                if not metal:
                    continue

                # 添加进口数据点
                if import_value > 0:
                    data_points.append(MetalDataPoint(
                        metal=metal,
                        data_type="import",
                        value=import_value,
                        unit="ton",
                        date=date,
                        source=self.name,
                        region="China",
                        metadata={"product_name": product_name}
                    ))

                # 添加出口数据点
                if export_value > 0:
                    data_points.append(MetalDataPoint(
                        metal=metal,
                        data_type="export",
                        value=export_value,
                        unit="ton",
                        date=date,
                        source=self.name,
                        region="China",
                        metadata={"product_name": product_name}
                    ))

            except Exception as e:
                print(f"[{self.name}] 解析行数据失败: {e}")
                continue

        return data_points

    def _parse_number(self, text: str) -> float:
        """解析数字（处理千位分隔符等）"""
        # 移除非数字字符（保留小数点）
        text = re.sub(r'[^\d.]', '', text)
        try:
            return float(text)
        except ValueError:
            return 0.0

    def _parse_date(self, date_str: str) -> datetime:
        """解析日期"""
        # 处理不同日期格式
        formats = [
            "%Y-%m",
            "%Y年%m月",
            "%Y/%m",
            "%Y-%m-%d",
        ]

        for fmt in formats:
            try:
                return datetime.strptime(date_str, fmt)
            except ValueError:
                continue

        # 如果都失败，返回当前日期
        return datetime.now()

    def _identify_metal(self, product_name: str) -> str:
        """
        从商品名称识别金属类型

        Args:
            product_name: 商品名称（如"铜及铜制品"）

        Returns:
            金属标识（如"copper"），未识别返回None
        """
        metal_mapping = {
            "铜": "copper",
            "铝": "aluminum",
            "锌": "zinc",
            "镍": "nickel",
            "铅": "lead",
            "锡": "tin",
```

```python
"""
金属数据存储模型
"""

from dataclasses import dataclass
from datetime import datetime
from typing import List, Optional, Dict
import sqlite3
import json


@dataclass
class MetalData:
    """金属数据模型"""
    id: Optional[int] = None
    metal: str = ""              # 金属类型
    data_type: str = ""          # 数据类型（import/export/production等）
    value: float = 0.0           # 数值
    unit: str = "ton"            # 单位
    date: datetime = None        # 数据日期
    source: str = ""             # 数据源
    region: str = "China"        # 地区
    metadata: Dict = None        # 元数据（JSON）
    created_at: datetime = None  # 记录创建时间


class MetalDataStorage:
    """金属数据存储类"""

    def __init__(self, db_path: str):
        self.db_path = db_path
        self._init_database()

    def _init_database(self):
        """初始化数据库表"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()

        # 创建金属数据表
        cursor.execute("""
        CREATE TABLE IF NOT EXISTS metal_data (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            metal TEXT NOT NULL,
            data_type TEXT NOT NULL,
            value REAL NOT NULL,
            unit TEXT NOT NULL,
            date TEXT NOT NULL,
            source TEXT NOT NULL,
            region TEXT DEFAULT 'China',
            metadata TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
        """)

        # 创建索引
        cursor.execute("""
        CREATE INDEX IF NOT EXISTS idx_metal_data_date
        ON metal_data(metal, data_type, date)
        """)

        cursor.execute("""
        CREATE INDEX IF NOT EXISTS idx_metal_data_source
        ON metal_data(source)
        """)

        conn.commit()
        conn.close()

    def save(self, data_points: List[MetalData]):
        """
        批量保存数据

        Args:
            data_points: 数据点列表
        """
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()

        for dp in data_points:
            # 检查是否已存在（去重）
            cursor.execute("""
            SELECT id FROM metal_data
            WHERE metal=? AND data_type=? AND date=? AND source=?
            """, (dp.metal, dp.data_type, dp.date.strftime("%Y-%m-%d"), dp.source))

            existing = cursor.fetchone()

            metadata_json = json.dumps(dp.metadata) if dp.metadata else None

            if existing:
                # 更新
                cursor.execute("""
                UPDATE metal_data
                SET value=?, unit=?, region=?, metadata=?
                WHERE id=?
                """, (dp.value, dp.unit, dp.region, metadata_json, existing[0]))
            else:
                # 插入
                cursor.execute("""
                INSERT INTO metal_data
                (metal, data_type, value, unit, date, source, region, metadata)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?)
                """, (
                    dp.metal,
                    dp.data_type,
                    dp.value,
                    dp.unit,
                    dp.date.strftime("%Y-%m-%d"),
                    dp.source,
                    dp.region,
                    metadata_json
                ))

        conn.commit()
        conn.close()
        print(f"✓ 已保存 {len(data_points)} 条金属数据")

    def query(
        self,
        metal: str,
        data_type: str,
        start_date: datetime,
        end_date: datetime,
    ) -> List[MetalData]:
        """
        查询数据

        Args:
            metal: 金属类型
            data_type: 数据类型
            start_date: 开始日期
            end_date: 结束日期

        Returns:
            数据点列表
        """
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()

        cursor.execute("""
        SELECT * FROM metal_data
        WHERE metal=? AND data_type=?
        AND date BETWEEN ? AND ?
        ORDER BY date DESC
        """, (
            metal,
            data_type,
            start_date.strftime("%Y-%m-%d"),
            end_date.strftime("%Y-%m-%d")
        ))

        rows = cursor.fetchall()
        conn.close()

        # 转换为对象
        data_points = []
        for row in rows:
            metadata = json.loads(row[8]) if row[8] else None
            dp = MetalData(
                id=row[0],
                metal=row[1],
                data_type=row[2],
                value=row[3],
                unit=row[4],
                date=datetime.strptime(row[5], "%Y-%m-%d"),
                source=row[6],
                region=row[7],
                metadata=metadata,
                created_at=datetime.fromisoformat(row[9]) if row[9] else None
            )
            data_points.append(dp)

        return data_points

    def get_latest(self, metal: str, data_type: str) -> Optional[MetalData]:
        """
        获取最新数据

        Args:
            metal: 金属类型
            data_type: 数据类型

        Returns:
            最新数据点
        """
        results = self.query(
            metal=metal,
            data_type=data_type,
            start_date=datetime(2000, 1, 1),
            end_date=datetime.now()
        )

        return results[0] if results else None
```

```python
"""

from typing import Dict, List, Tuple

@dataclass
class SupplyDemandBalance:
    """供需平衡结果"""
    metal: str
    date: datetime

    # 供给侧
    production: float       # 产量
    imports: float          # 进口
    total_supply: float     # 总供给

    # 需求侧
    consumption: float      # 消费量
    exports: float          # 出口
    total_demand: float     # 总需求

    # 平衡
    balance: float          # 供需差额（正值=过剩，负值=短缺）
    balance_rate: float     # 供需比例

    # 库存
    inventory: float        # 总库存
    inventory_days: float   # 库存天数（库存/日均消费）

    # 结论
    status: str             # 平衡状态：surplus(过剩) / balanced(平衡) / shortage(短缺)
    risk_level: str         # 风险等级：low / medium / high


class SupplyDemandAnalyzer:
    """供需分析器"""

    def __init__(self, storage: MetalDataStorage):
        self.storage = storage

    def analyze(
        self,
        metal: str,
        date: datetime,
        lookback_days: int = 90
    ) -> SupplyDemandBalance:
        """
        分析指定金属的供需平衡

        Args:
            metal: 金属类型（如copper）
            date: 分析日期
            lookback_days: 回溯天数

        Returns:
            供需平衡分析结果
        """
        start_date = date - timedelta(days=lookback_days)

        # 1. 获取供给侧数据
        production = self._get_sum_value(metal, "production", start_date, date)
        imports = self._get_sum_value(metal, "import", start_date, date)
        total_supply = production + imports

        # 2. 获取需求侧数据
        consumption = self._get_sum_value(metal, "consumption", start_date, date)
        exports = self._get_sum_value(metal, "export", start_date, date)
        total_demand = consumption + exports

        # 3. 计算平衡
        balance = total_supply - total_demand
        balance_rate = (total_supply / total_demand * 100) if total_demand > 0 else 0

        # 4. 获取库存数据
        inventory = self._get_latest_value(metal, "inventory", date)
        daily_consumption = consumption / lookback_days if consumption > 0 else 0
        inventory_days = inventory / daily_consumption if daily_consumption > 0 else 0

        # 5. 判断状态
        status = self._determine_status(balance_rate)
        risk_level = self._determine_risk_level(balance_rate, inventory_days)

        return SupplyDemandBalance(
            metal=metal,
            date=date,
            production=production,
            imports=imports,
            total_supply=total_supply,
            consumption=consumption,
            exports=exports,
            total_demand=total_demand,
            balance=balance,
            balance_rate=balance_rate,
            inventory=inventory,
            inventory_days=inventory_days,
            status=status,
            risk_level=risk_level
        )

    def _get_sum_value(
        self,
        metal: str,
        data_type: str,
        start_date: datetime,
        end_date: datetime
    ) -> float:
        """获取指定时间段的数据总和"""
        data_points = self.storage.query(metal, data_type, start_date, end_date)
        return sum(dp.value for dp in data_points)

    def _get_latest_value(self, metal: str, data_type: str, before_date: datetime) -> float:
        """获取最新数据"""
        data_point = self.storage.get_latest(metal, data_type)
        return data_point.value if data_point else 0.0

    def _determine_status(self, balance_rate: float) -> str:
        """判断供需状态"""
        if balance_rate > 105:
            return "surplus"      # 过剩
        elif balance_rate < 95:
            return "shortage"     # 短缺
        else:
            return "balanced"     # 平衡

    def _determine_risk_level(self, balance_rate: float, inventory_days: float) -> str:
        """判断风险等级"""
        # 综合考虑供需比例和库存天数
        if balance_rate < 90 or inventory_days < 20:
            return "high"         # 高风险（供应紧张或库存低）
        elif balance_rate > 110 or inventory_days > 60:
            return "high"         # 高风险（供应过剩或库存高）
        elif 95 <= balance_rate <= 105 and 30 <= inventory_days <= 45:
            return "low"          # 低风险（供需平衡且库存合理）
        else:
            return "medium"       # 中风险

    def generate_report(self, result: SupplyDemandBalance) -> str:

        }
```

```python
"""
异常行为检测器
"""

from typing import Dict, List, Optional
from datetime import datetime, timedelta
from dataclasses import dataclass
import numpy as np

from trendradar.storage.metals import MetalDataStorage


@dataclass
class AnomalyAlert:
    """异常告警"""
    metal: str
    alert_type: str         # hoarding(囤货) / squeeze(逼仓) / surge(激增)
    severity: str           # low / medium / high / critical
    date: datetime
    indicators: Dict        # 指标数据
    description: str        # 描述
    recommendation: str     # 建议


class AnomalyDetector:
    """异常检测器"""

    def __init__(
        self,
        storage: MetalDataStorage,
        config: Dict
    ):
        self.storage = storage
        self.config = config

    def detect_hoarding(
        self,
        metal: str,
        date: datetime,
        lookback_months: int = 6
    ) -> Optional[AnomalyAlert]:
        """
        检测囤货行为

        囤货特征：
        1. 库存异常增长（远超历史均值）
        2. 进口激增但消费未增
        3. 价格上涨但库存增加（正常情况应减少）

        Args:
            metal: 金属类型
            date: 检测日期
            lookback_months: 回溯月数

        Returns:
            异常告警（如果检测到）
        """
        # 1. 获取库存数据
        inventory_history = self._get_time_series(
            metal, "inventory", date, lookback_months
        )

        if not inventory_history:
            return None

        # 2. 计算库存增长率
        current_inventory = inventory_history[-1]
        avg_inventory = np.mean(inventory_history[:-1])
        std_inventory = np.std(inventory_history[:-1])

        growth_rate = ((current_inventory - avg_inventory) / avg_inventory * 100) if avg_inventory > 0 else 0
        z_score = ((current_inventory - avg_inventory) / std_inventory) if std_inventory > 0 else 0

        # 3. 获取进口数据
        recent_imports = self._get_sum_in_period(
            metal, "import", date, days=30
        )
        avg_imports = self._get_avg_in_period(
            metal, "import", date - timedelta(days=180), date - timedelta(days=30), period_days=30
        )

        import_surge = ((recent_imports - avg_imports) / avg_imports * 100) if avg_imports > 0 else 0

        # 4. 获取价格数据
        price_history = self._get_time_series(
            metal, "spot_price", date, lookback_months=3
        )

        price_change = 0
        if len(price_history) >= 2:
            price_change = ((price_history[-1] - price_history[0]) / price_history[0] * 100)

        # 5. 判断是否异常
        threshold = self.config.get("hoarding_detection", {}).get("inventory_growth_threshold", 20)

        is_hoarding = False
        severity = "low"

        # 规则1：库存异常增长
        if growth_rate > threshold and z_score > 2:
            is_hoarding = True
            severity = "high" if z_score > 3 else "medium"

        # 规则2：进口激增 + 库存增加
        if import_surge > 30 and growth_rate > 15:
            is_hoarding = True
            severity = "high"

        # 规则3：价格上涨 + 库存增加（背离）
        if price_change > 10 and growth_rate > 15:
            is_hoarding = True
            severity = "critical" if price_change > 20 else "high"

        if not is_hoarding:
            return None

        # 6. 生成告警
        description = self._generate_hoarding_description(
            growth_rate, import_surge, price_change, z_score
        )

        recommendation = self._generate_hoarding_recommendation(severity)

        return AnomalyAlert(
            metal=metal,
            alert_type="hoarding",
            severity=severity,
            date=date,
            indicators={
                "inventory_growth": f"{growth_rate:.1f}%",
                "import_surge": f"{import_surge:.1f}%",
                "price_change": f"{price_change:.1f}%",
                "z_score": f"{z_score:.2f}",
                "current_inventory": f"{current_inventory:,.0f} 吨",
                "avg_inventory": f"{avg_inventory:,.0f} 吨"
            },
            description=description,
            recommendation=recommendation
        )

    def detect_squeeze(
        self,
        metal: str,
        date: datetime
    ) -> Optional[AnomalyAlert]:
        """
        检测逼仓行为

        逼仓特征：
        1. 现货大幅升水（backwardation）
        2. 交易所库存急剧下降
        3. 持仓集中度高

        Args:
            metal: 金属类型
            date: 检测日期

        Returns:
            异常告警（如果检测到）
        """
        # 1. 获取现货与期货价差
        spot_price = self._get_latest_value(metal, "spot_price", date)
        future_price = self._get_latest_value(metal, "future_price", date)

        if not spot_price or not future_price:
            return None

        premium = spot_price - future_price  # 升水（正值）或贴水（负值）

        # 2. 获取交易所库存变化
        inventory_history = self._get_time_series(
            metal, "exchange_inventory", date, lookback_months=3
        )

        if len(inventory_history) < 2:
            return None

        inventory_change = inventory_history[-1] - inventory_history[0]
        inventory_change_rate = (inventory_change / inventory_history[0] * 100) if inventory_history[0] > 0 else 0

        # 3. 判断是否逼仓
        threshold = self.config.get("squeeze_detection", {}).get("backwardation_threshold", 200)

        is_squeeze = False
        severity = "low"

        # 规则：现货大幅升水 + 库存急剧下降
        if premium > threshold and inventory_change_rate < -20:
            is_squeeze = True
            severity = "critical" if premium > threshold * 2 else "high"

        if not is_squeeze:
            return None

        # 4. 生成告警
        description = f"""
检测到{metal}市场可能存在逼仓风险：

- 现货升水达到 {premium:.0f} 元/吨，远超正常水平
- 交易所库存下降 {-inventory_change_rate:.1f}%
- 市场流动性紧张，空头可能面临交割压力
"""

        recommendation = """
建议：
1. 密切关注持仓报告，警惕大户集中持仓
2. 关注交割日临近时的仓单变化
3. 空头需提前准备交割货源或考虑平仓
"""

        return AnomalyAlert(
            metal=metal,
            alert_type="squeeze",
            severity=severity,
            date=date,
            indicators={
                "spot_premium": f"{premium:.0f} 元/吨",
                "inventory_change": f"{inventory_change_rate:.1f}%",
                "spot_price": f"{spot_price:.0f} 元/吨",
                "future_price": f"{future_price:.0f} 元/吨"
            },
            description=description,
            recommendation=recommendation
        )

    def _get_time_series(
        self,
        metal: str,
        data_type: str,
        end_date: datetime,
        lookback_months: int
    ) -> List[float]:
        """获取时间序列数据"""
        start_date = end_date - timedelta(days=lookback_months * 30)
        data_points = self.storage.query(metal, data_type, start_date, end_date)
        return [dp.value for dp in data_points]

    def _get_sum_in_period(
        self,
        metal: str,
        data_type: str,
        end_date: datetime,
        days: int
    ) -> float:
        """获取指定时段的数据总和"""
        start_date = end_date - timedelta(days=days)
        data_points = self.storage.query(metal, data_type, start_date, end_date)
        return sum(dp.value for dp in data_points)

    def _get_avg_in_period(
        self,
        metal: str,
        data_type: str,
        start_date: datetime,
        end_date: datetime,
        period_days: int
    ) -> float:
        """获取平均值"""
        data_points = self.storage.query(metal, data_type, start_date, end_date)
        if not data_points:
            return 0.0
        return sum(dp.value for dp in data_points) / len(data_points)

    def _get_latest_value(self, metal: str, data_type: str, before_date: datetime) -> Optional[float]:
        """获取最新值"""
        data_point = self.storage.get_latest(metal, data_type)
        return data_point.value if data_point else None

    def _generate_hoarding_description(
        self,
        growth_rate: float,
        import_surge: float,
        price_change: float,
        z_score: float
    ) -> str:
        """生成囤货描述"""
        desc = f"检测到库存异常增长 {growth_rate:.1f}%（{z_score:.2f}倍标准差）"

        if import_surge > 30:
            desc += f"，同时进口量激增 {import_surge:.1f}%"

        if price_change > 10:
            desc += f"，价格上涨 {price_change:.1f}% 但库存不降反增"

        desc += "。此现象可能表明市场存在囤货行为。"

        return desc

    def _generate_hoarding_recommendation(self, severity: str) -> str:
        """生成囤货建议"""
        if severity == "critical":
            return """
建议：
1. 高度关注库存去化速度
2. 警惕囤货释放可能带来的价格下跌
3. 考虑适当减仓或对冲风险
"""
        elif severity == "high":
            return """
建议：
1. 持续跟踪库存变化
2. 关注下游消费数据
3. 谨慎评估价格上行空间
"""
        else:
            return """
建议：
1. 保持观察
2. 关注后续数据变化
"""
```

### 3. 新闻与数据关联：`trendradar/analyzer/correlation.py`

```python
"""
新闻与数据关联分析器
"""

from typing import List, Dict
from datetime import datetime, timedelta

from trendradar.storage.base import NewsItem
from trendradar.storage.metals import MetalDataStorage
from trendradar.analyzer.supply_demand import SupplyDemandAnalyzer
from trendradar.analyzer.anomaly import AnomalyDetector


class NewsDataCorrelator:
    """新闻与数据关联分析器"""

    def __init__(
        self,
        metal_storage: MetalDataStorage,
        config: Dict
    ):
        self.metal_storage = metal_storage
        self.supply_demand_analyzer = SupplyDemandAnalyzer(metal_storage)
        self.anomaly_detector = AnomalyDetector(metal_storage, config)

    def analyze_news_with_data(
        self,
        news_item: NewsItem,
        date: datetime
    ) -> Dict:
        """
        分析单条新闻并关联相关数据

        Args:
            news_item: 新闻条目
            date: 分析日期

        Returns:
            关联分析结果
        """
        # 1. 识别新闻中的金属类型
        metals = self._identify_metals(news_item.title)

        if not metals:
            return {}

        result = {
            "news": {
                "title": news_item.title,
                "platform": news_item.platform_name,
                "url": news_item.url
            },
            "metals_analysis": {}
        }

        # 2. 对每种金属进行分析
        for metal in metals:
            metal_analysis = {
                "supply_demand": None,
                "anomalies": [],
                "recent_data": {},
                "insights": []
            }

            # 供需分析
            try:
                supply_demand = self.supply_demand_analyzer.analyze(metal, date)
                metal_analysis["supply_demand"] = supply_demand
            except Exception as e:
                print(f"供需分析失败: {e}")

            # 异常检测
            try:
                hoarding_alert = self.anomaly_detector.detect_hoarding(metal, date)
                if hoarding_alert:
                    metal_analysis["anomalies"].append(hoarding_alert)

                squeeze_alert = self.anomaly_detector.detect_squeeze(metal, date)
                if squeeze_alert:
                    metal_analysis["anomalies"].append(squeeze_alert)
            except Exception as e:
                print(f"异常检测失败: {e}")

            # 获取最近数据
            metal_analysis["recent_data"] = self._get_recent_data(metal, date)

            # 生成洞察
            metal_analysis["insights"] = self._generate_insights(
                news_item.title,
                metal_analysis
            )

            result["metals_analysis"][metal] = metal_analysis

        return result

    def _identify_metals(self, title: str) -> List[str]:
        """从标题中识别金属类型"""
        metal_keywords = {
            "copper": ["铜", "copper", "Cu"],
            "aluminum": ["铝", "aluminum", "Al"],
            "zinc": ["锌", "zinc", "Zn"],
            "nickel": ["镍", "nickel", "Ni"],
        }

        identified = []
        for metal, keywords in metal_keywords.items():
            if any(keyword in title for keyword in keywords):
                identified.append(metal)

        return identified

    def _get_recent_data(self, metal: str, date: datetime) -> Dict:
        """获取最近数据摘要"""
        recent = {}

        # 最近7天的进口数据
        import_7d = self.metal_storage.query(
            metal, "import",
            date - timedelta(days=7), date
        )
        if import_7d:
            recent["import_7d"] = sum(dp.value for dp in import_7d)

        # 最新库存
        inventory = self.metal_storage.get_latest(metal, "inventory")
        if inventory:
            recent["inventory"] = inventory.value

        # 最新价格
        price = self.metal_storage.get_latest(metal, "spot_price")
        if price:
            recent["price"] = price.value

        return recent

    def _generate_insights(self, title: str, analysis: Dict) -> List[str]:
        """
        生成洞察分析

        根据新闻标题和数据分析结果，生成关联洞察
        """
        insights = []

        # 分析新闻关键词
        is_price_rise = any(keyword in title for keyword in ["涨", "上涨", "飙升", "大涨"])
        is_price_fall = any(keyword in title for keyword in ["跌", "下跌", "暴跌", "大跌"])

        supply_demand = analysis.get("supply_demand")
        anomalies = analysis.get("anomalies", [])

        # 洞察1：价格与供需的关系
        if supply_demand:
            if is_price_rise:
                if supply_demand.status == "shortage":
                    insights.append(
                        f"✓ 价格上涨与数据吻合：当前供应短缺（供需比{supply_demand.balance_rate:.1f}%），"
                        f"需求超出供给约{-supply_demand.balance:.0f}吨"
                    )
                elif supply_demand.status == "surplus":
                    insights.append(
                        f"⚠️ 价格与供需背离：价格上涨但供应过剩（供需比{supply_demand.balance_rate:.1f}%），"
                        f"可能存在投机炒作或其他因素"
                    )

            elif is_price_fall:
                if supply_demand.status == "surplus":
                    insights.append(
                        f"✓ 价格下跌与数据吻合：当前供应过剩（供需比{supply_demand.balance_rate:.1f}%），"
                        f"供给超出需求约{supply_demand.balance:.0f}吨"
                    )
                elif supply_demand.status == "shortage":
                    insights.append(
                        f"⚠️ 价格与供需背离：价格下跌但供应短缺，需关注需求端是否出现问题"
                    )

        # 洞察2：异常行为
        for anomaly in anomalies:
            if anomaly.alert_type == "hoarding":
                insights.append(
                    f"🔴 检测到囤货风险：{anomaly.description}"
                )
            elif anomaly.alert_type == "squeeze":
                insights.append(
                    f"🔴 检测到逼仓风险：{anomaly.description}"
                )

        # 洞察3：库存天数评估
        if supply_demand and supply_demand.inventory_days < 20:
            insights.append(
                f"⚠️ 库存紧张：当前库存仅够{supply_demand.inventory_days:.1f}天消费，"
                f"低于安全水平（通常>30天）"
            )

        return insights
```

### 4. AI深度分析集成：`trendradar/ai/metals_analyzer.py`

```python
"""
金属行业AI分析器（扩展原有AI模块）
"""

from typing import Dict, List
from datetime import datetime

from trendradar.ai.client import AIClient
from trendradar.analyzer.correlation import NewsDataCorrelator


class MetalsAIAnalyzer:
    """金属行业AI分析器"""

    def __init__(
        self,
        ai_client: AIClient,
        correlator: NewsDataCorrelator
    ):
        self.ai_client = ai_client
        self.correlator = correlator

    def analyze_with_data(
        self,
        news_items: List,
        date: datetime
    ) -> str:
        """
        结合新闻和数据进行AI分析

        Args:
            news_items: 新闻列表
            date: 分析日期

        Returns:
            AI分析报告（Markdown格式）
        """
        # 1. 对每条新闻进行数据关联
        correlations = []
        for news in news_items:
            correlation = self.correlator.analyze_news_with_data(news, date)
            if correlation:
                correlations.append(correlation)

        # 2. 构建AI分析提示词
        prompt = self._build_prompt(news_items, correlations)

        # 3. 调用AI生成分析
        analysis = self.ai_client.analyze(prompt)

        return analysis

    def _build_prompt(self, news_items: List, correlations: List[Dict]) -> str:
        """构建AI分析提示词"""

        # 整理新闻
        news_section = "## 今日新闻\n\n"
        for i, news in enumerate(news_items, 1):
            news_section += f"{i}. 【{news.platform_name}】{news.title}\n"

        # 整理数据关联
        data_section = "\n## 数据关联分析\n\n"
        for corr in correlations:
            news_title = corr["news"]["title"]
            data_section += f"### {news_title}\n\n"

            for metal, analysis in corr["metals_analysis"].items():
                data_section += f"#### {metal.upper()}\n\n"

                # 供需数据
                sd = analysis.get("supply_demand")
                if sd:
                    data_section += f"""
**供需平衡**:
- 供给: {sd.total_supply:,.0f}吨（产量{sd.production:,.0f} + 进口{sd.imports:,.0f}）
- 需求: {sd.total_demand:,.0f}吨（消费{sd.consumption:,.0f} + 出口{sd.exports:,.0f}）
- 平衡: {sd.balance:+,.0f}吨（{sd.status}）
- 库存: {sd.inventory:,.0f}吨（{sd.inventory_days:.1f}天）
"""

                # 异常告警
                anomalies = analysis.get("anomalies", [])
                if anomalies:
                    data_section += "\n**异常告警**:\n"
                    for anomaly in anomalies:
                        data_section += f"- {anomaly.alert_type}: {anomaly.description}\n"

                # 洞察
                insights = analysis.get("insights", [])
                if insights:
                    data_section += "\n**关联洞察**:\n"
                    for insight in insights:
                        data_section += f"- {insight}\n"

                data_section += "\n"

        # 构建完整提示词
        prompt = f"""
你是一位资深的金属行业分析师。请基于以下新闻和数据，进行深度分析：

{news_section}

{data_section}

请从以下角度进行分析（2000字左右）：

1. **核心观点总结**：当前金属市场的主要变化是什么？

2. **供给侧分析**：
   - 矿山产量、冶炼产能、进口情况如何？
   - 供给端有哪些变化或风险？

3. **需求侧分析**：
   - 下游消费（电力、家电、汽车、建筑等）表现如何？
   - 需求端有哪些亮点或隐忧？

4. **库存与囤货**：
   - 库存水平是否合理？
   - 是否存在囤货、逼仓等异常行为？
   - 对价格有何影响？

5. **价格驱动因素**：
   - 本次价格变化的主要驱动力是什么（供给/需求/投机/宏观）？
   - 供需基本面与价格是否匹配？

6. **风险与机会**：
   - 当前市场面临的主要风险是什么？
   - 是否存在投资或套利机会？

7. **后市展望**：
   - 预计短期（1-3个月）和中期（3-6个月）价格走势？
   - 需要关注哪些关键数据和事件？

请结合具体数据，进行专业、客观的分析。
"""

        return prompt
```
